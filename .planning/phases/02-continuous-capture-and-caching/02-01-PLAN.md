    **TranslationCache.kt:**
    Two-tier cache with `android.util.LruCache` in memory fronting Room. Per locked decision: cache key is Japanese source text (exact string match), cache value is English translation + metadata.

    ```kotlin
    class TranslationCache(
        private val cachedTranslationDao: CachedTranslationDao,
        maxMemoryEntries: Int = 500
    ) {
        private val memoryCache = LruCache<String, String>(maxMemoryEntries)

        suspend fun get(japaneseText: String): String? {
            // Tier 1: In-memory LRU
            memoryCache.get(japaneseText)?.let { return it }

            // Tier 2: Room database
            val dbEntry = cachedTranslationDao.findBySourceText(japaneseText)
            if (dbEntry != null) {
                // Promote to memory cache
                memoryCache.put(japaneseText, dbEntry.translatedText)
                return dbEntry.translatedText
            }

            return null // Cache miss
        }

        suspend fun put(japaneseText: String, englishText: String) {
            memoryCache.put(japaneseText, englishText)
            cachedTranslationDao.insert(
                CachedTranslationEntity(
                    sourceText = japaneseText,
                    translatedText = englishText,
                    timestamp = System.currentTimeMillis()
                )
            )
        }

        suspend fun clear() {
            memoryCache.evictAll()
            cachedTranslationDao.deleteAll()
        }
    }
    ```

    **Extend TranslationManager with cache integration:**

    Add `TranslationCache` as a constructor dependency (Hilt-injected). Modify `translate()` to check cache before calling the API:

    ```kotlin
    @Singleton
    class TranslationManager @Inject constructor(
        private val deepLEngine: DeepLTranslationEngine,
        private val mlKitEngine: MlKitTranslationEngine,
        private val settingsRepository: SettingsRepository,
        private val translationCache: TranslationCache  // NEW
    ) {
        suspend fun translate(text: String): String {
            // Check cache first
            translationCache.get(text)?.let { cached ->
                return cached
            }

            // Cache miss -- call translation API (existing fallback chain)
            val result = translateViaApi(text)

            // Store in cache if translation succeeded (not an error message)
            if (!result.startsWith("Translation failed:")) {
                translationCache.put(text, result)
            }

            return result
        }

        // Extract existing translate logic into translateViaApi
        private suspend fun translateViaApi(text: String): String {
            val apiKey = settingsRepository.getDeepLApiKey()
            if (apiKey.isNullOrBlank()) {
                return translateWithMlKit(text)
            }
            return try {
                deepLEngine.translate(text)
            } catch (e: Exception) {
                translateWithMlKit(text)
            }
        }

        private suspend fun translateWithMlKit(text: String): String {
            return try {
                mlKitEngine.translate(text)
            } catch (e: Exception) {
                "Translation failed: ${e.message}"
            }
        }

        /** Clear the translation cache (both in-memory and persistent). */
        suspend fun clearCache() {
            translationCache.clear()
        }
    }
    ```

    Also update `di/TranslationModule.kt` if needed -- TranslationManager is already provided via @Inject constructor, so adding TranslationCache to its constructor should auto-resolve via Hilt since TranslationCache is provided in AppModule.
